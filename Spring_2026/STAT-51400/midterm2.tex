\section*{1) Box--Cox Transformations}
For response $Y>0$ and parameter $\lambda$:
\[
Y^{(\lambda)}=
\begin{cases}
\dfrac{Y^\lambda-1}{\lambda}, & \lambda\neq0,\\
\ln(Y), & \lambda=0.
\end{cases}
\]
If data include zero/negative values, use offset $Y+c$ with $c> -\min(Y)$.

\textbf{When to use:}
\begin{itemize}
  \item Residuals are skewed (normality fails).
  \item Variability changes with mean (constant variance fails).
\end{itemize}

\textbf{Why use:}
\begin{itemize}
  \item Stabilize variance (Make it constant).
  \item Make residuals closer to normal.
  \item Improve validity of ANOVA $F$ tests.
\end{itemize}

\textbf{Common variance-stabilizing transformations}
\begin{center}
\scriptsize
\begin{tabular}{llll}
\toprule
$\sigma_y$ vs $\mu$ & $a$ & $\lambda=1-a$ & Transformation \\
\midrule
constant & $0$ & $1$ & none \\
$\propto \mu^{1/2}$ & $1/2$ & $1/2$ & square root \\
$\propto \mu$ & $1$ & $0$ & log \\
$\propto \mu^{3/2}$ & $3/2$ & $-1/2$ & reciprocal sqrt \\
$\propto \mu^2$ & $2$ & $-1$ & reciprocal \\
\bottomrule
\end{tabular}
\end{center}

\section*{2) Randomization Test}
Use when the usual ANOVA reference distribution is unreliable (Normality fails).

Hypotheses (one-way ANOVA setting):
\begin{align*}
H_0&:\ \mu_1=\cdots=\mu_a, \\
H_1&:\ \text{not all means are equal}
\end{align*}

For $a$ levels and $n$ replicates each ($N=an$):
\[
K=\frac{N!}{(n!)^a}
\]
possible rearrangements under $H_0$.

For each rearrangement compute an $F$ statistic. Let $C$ be the number with $F\ge F_0$.

\[
F_0 := \text{Initial F-statistic}
\]

Decision rule:
\[
\frac{C}{K}<\alpha\ \Rightarrow\ \text{reject }H_0.
\]

\section*{3) Fisher's LSD}
Given the hypothesis test:
\begin{align*}
    H_0&: \mu_i=\mu_j\\
    H_1&:\mu_i \neq \mu_j
\end{align*}
Compute $LSD$ statistic:
\[
\mathrm{LSD}=t_{\alpha/2,\,N- a}\sqrt{\frac{2MS_E}{n}}
\]
Then reject null hypothesis if:
\[
|\bar y_i-\bar y_j|>\mathrm{LSD}\ \Rightarrow\ \mu_i\neq\mu_j
\]
Equivalent test statistic:
\[
t_0=\frac{\bar y_{i\cdot}-\bar y_{j\cdot}}{\sqrt{MS_E\left(\frac1{n_i}+\frac1{n_j}\right)}}
\]
LSD is more sensitive, but more likely to declare a false difference when many pairs are tested.
Fisher LSD inflates type-I error
\section*{4) Tukey's Test}
Given the hypothesis test:
\begin{align*}
    H_0&: \mu_i=\mu_j\\
    H_1&:\mu_i \neq \mu_j
\end{align*}
Compute $HSD$ statistic:
\[
\mathrm{HSD}=q_{\alpha;\,a,\,df_E}\sqrt{\frac{MSE}{n}}
\]
Then reject null hypothesis if:
\[
|\bar y_i-\bar y_j|>\mathrm{HSD}\ \Rightarrow\ \mu_i\neq\mu_j
\]
Tukey is safer when comparing many pairs because it controls the overall probability of at least one false positive.


\section*{5) Latin Square Design (LSD)}
Controls two nuisance factors (rows, columns) plus treatment.

Model:
\[
y_{ijk}=\mu+\tau_j+\alpha_i+\beta_k+\varepsilon_{ijk},\quad i,j,k=1,\dots,p
\]

ANOVA sum-of-squares partition:
\[
SS_T=SS_{\text{Treatments}}+SS_{\text{Rows}}+SS_{\text{Cols}}+SS_E
\]
\[
SS_T=\sum_{i,j,k}(y_{ijk}-\bar y_{...})^2
\]
\[
SS_{\text{Treatments}}=p\sum_{j=1}^{p}(\bar y_{.j.}-\bar y_{...})^2
\]
\[
SS_{\text{Rows}}=p\sum_{i=1}^{p}(\bar y_{i..}-\bar y_{...})^2
\]
\[
SS_{\text{Cols}}=p\sum_{k=1}^{p}(\bar y_{..k}-\bar y_{...})^2
\]
\textbf{ANOVA table (Latin square)}
\begin{center}
\footnotesize
\begin{tabular}{llll}
\toprule
Source & df & Mean Square & $F_0$ \\
\midrule
Treatments & $p-1$ & $MS_{\text{Trt}}$ & $MS_{\text{Trt}}/MS_E$ \\
Rows & $p-1$ & $MS_{\text{Rows}}$ & -- \\
Columns & $p-1$ & $MS_{\text{Cols}}$ & --\\
Error & $(p-1)(p-2)$ & $MS_E$ & -- \\
Total & $p^2-1$ & -- & -- \\
\bottomrule
\end{tabular}
\end{center}

\section*{6) Graeco-Latin Square (GLSD)}
Controls three nuisance sources via superposed orthogonal Latin squares.

Model:
\[
y_{ijkl}=\mu+L_j+G_k+R_i+C_l+\varepsilon_{ijkl}
\]

ANOVA partition:
\[
SS_T=SS_L+SS_G+SS_R+SS_C+SS_E
\]
\[
SS_T=\sum_{i,j,k,l}(y_{ijkl}-\bar y_{....})^2
\]
\[
SS_L=p\sum_{j=1}^{p}(\bar y_{.j..}-\bar y_{....})^2,
\quad
SS_G=p\sum_{k=1}^{p}(\bar y_{..k.}-\bar y_{....})^2
\]
\[
SS_R=p\sum_{i=1}^{p}(\bar y_{i...}-\bar y_{....})^2,
\quad
SS_C=p\sum_{l=1}^{p}(\bar y_{...l}-\bar y_{....})^2
\]

\textbf{ANOVA table (Graeco-Latin square)}
\begin{center}
\footnotesize
\begin{tabular}{llll}
\toprule
Source & df & Mean Square & $F_0$ \\
\midrule
Latin (Trt) & $p-1$ & $MS_L$ & $MS_L/MS_E$ \\
Greek & $p-1$ & $MS_G$ & -- \\
Rows & $p-1$ & $MS_R$ & --\\
Columns & $p-1$ & $MS_C$ & -- \\
Error & $(p-1)(p-3)$ & $MS_E$ & -- \\
Total & $p^2-1$ & -- & -- \\
\bottomrule
\end{tabular}
\end{center}

\section*{6) BIBD}
Parameters:
7\begin{align*}
a=&\#\text{treatments} \\ b=&\#\text{blocks}\\ k=&\text{block size}\\ r=&\text{replications/treatment}
\end{align*}
\[
N=ar=bk,\qquad
\lambda=\frac{r(k-1)}{a-1}\in\mathbb{Z}
\]
Balance: each treatment appears $r$ times; each pair appears together $\lambda$ times.

\[
y_{ij} = \mu + \tau_i + \beta_j + \varepsilon_{ij}
\]

\[
SS_T = SS_{\text{Treatments(adjusted)}} + SS_{\text{Blocks}} + SS_E
\]

\[
SS_{\text{Total}} =
\sum_{i=1}^{a}\sum_{j=1}^{b}(y_{ij}-\bar{y}_{..})^2
\]

\[
SS_{\text{Treatments(adjusted)}} =
b\sum_{i=1}^{a}(\bar{y}_{i.}-\bar{y}_{..})^2
\]

\[
SS_{\text{Blocks}} =
a\sum_{j=1}^{b}(\bar{y}_{.j}-\bar{y}_{..})^2
\]
\[
SS_{\text{Total}} =
\sum_{i=1}^{a}\sum_{j=1}^{b} y_{ij}^2
- \frac{y_{..}^2}{N}
\]
\[
SS_{\text{Treatments(adjusted)}} =
\frac{1}{b}\sum_{i=1}^{a} y_{i.}^2
- \frac{y_{..}^2}{N}
\]
\[
SS_{\text{Blocks}} =
\frac{1}{a}\sum_{j=1}^{b} y_{.j}^2
- \frac{y_{..}^2}{N}
\]
\begin{center}
\footnotesize
\begin{tabular}{llll}
\toprule
Source & df & Mean Square & $F_0$ \\
\midrule
Treatments (Adj) & $a-1$ & $MS_{Trt}$ & $MS_{Trt}/MS_E$ \\
Blocks & $b-1$ & $MS_B$ & -- \\
Error & $N-a-b+1$ & $MS_E$ & -- \\
Total & $N-1$ & $MS_T$ & --\\
\bottomrule
\end{tabular}
\end{center}

\section*{7) Factorial Design ($2^k$) -- Quick Notes}
\textbf{Idea:} study $k$ factors simultaneously using all level combinations.
In a $2^k$ design, each factor has two levels: low ($-$) and high ($+$).

\textbf{Main effect:}
\[
\text{Effect of }A=\bar y_{A+}-\bar y_{A-}
\]
Interpretation: average change in response when $A$ moves from low to high.

\textbf{Interaction between factors:}
interaction exists when the effect of one factor depends on the level of another factor.
For $2^2$:
\[
AB=\frac{1}{2}\Big[(\bar y_{++}-\bar y_{-+})-(\bar y_{+-}-\bar y_{--})\Big]
\]
equivalently:
\[
AB=\frac{1}{2}\Big[(\bar y_{++}+\bar y_{--})-(\bar y_{+-}+\bar y_{-+})\Big]
\]

\textbf{Plots:}
\begin{itemize}
  \item \textit{Main effect plot}: means by factor level; steeper slope $\Rightarrow$ larger main effect.
  \item \textit{Interaction plot}: parallel lines $\Rightarrow$ no interaction; non-parallel/crossing lines $\Rightarrow$ interaction.
\end{itemize}

\textbf{ANOVA for factorial design:}
\[
y_{ijk}=\mu+\tau_i+\beta_j+(\tau\beta)_{ij}+\varepsilon_{ijk}
\]
\[
SS_T=SS_A+SS_B+SS_{AB}+SS_E
\]
In replicated $2^k$: each effect has 1 df, and
\[
df_E=2^k(n-1)
\]
Each effect is tested with $F_0=MS_{\text{effect}}/MS_E$.

\textbf{ANOVA table (general factorial design): degrees of freedom only}
\begin{center}
\footnotesize
\begin{tabular}{ll}
\toprule
Source & df \\
\midrule
Main effects ($r=1$) & $\binom{k}{1}(s-1)$ \\
Two-factor interactions ($r=2$) & $\binom{k}{2}(s-1)^2$ \\
$\vdots$ $r$-factor interactions & $\binom{k}{r}(s-1)^r$ \\
$k$-factor interaction ($r=k$) & $\binom{k}{k}(s-1)^k=(s-1)^k$ \\
Error & $N-\prod_{i=1}^{k}a_i$ \\
Total & $N-1$ \\
\bottomrule
\end{tabular}
\end{center}
Here $s$ is the common number of levels per factor (balanced $s^k$ factorial).

\textbf{Linear regression (coded variables):}
\[
y=\beta_0+\beta_1x_1+\beta_2x_2+\beta_{12}x_1x_2+\varepsilon,\quad x_i\in\{-1,+1\}
\]
ANOVA and regression are equivalent in $2^2$ designs.
With $\pm1$ coding, each coefficient is half of its corresponding effect:
\[
\hat A=2\hat\beta_1,\quad \hat B=2\hat\beta_2,\quad \widehat{AB}=2\hat\beta_{12}
\]
Least-squares estimator:
\[
\hat{\beta}=(X'X)^{-1}X'y
\]
In balanced $2^2$, the design is orthogonal, so $X'X$ is diagonal and estimation is straightforward.

\textbf{Prediction equation for $2^2$ (coded scale):}
\[
\hat{y}(x_1,x_2)=\hat{\beta}_0+\hat{\beta}_1x_1+\hat{\beta}_2x_2+\hat{\beta}_{12}x_1x_2
\]
Use this to compare factor settings and identify combinations that maximize/minimize response.

\textbf{Contrasts:}
\[
\Gamma=\sum_{i=1}^{2^k} c_i\bar y_i,\qquad \sum c_i=0
\]
\[
SS_\Gamma=\frac{\left(\sum c_i\bar y_i\right)^2}{\sum c_i^2/n}
\]
In $2^k$, orthogonal contrasts produce sums of squares for main effects and interactions.

\textbf{Synergy vs Antagonism (interaction):}
\begin{itemize}
  \item \textbf{Synergy} (positive interaction): combined effect is larger than additive expectation.
  \item \textbf{Antagonism} (negative interaction): combined effect is smaller than additive expectation.
  \item In interaction plots: diverging lines often indicate synergy; converging/crossing lines often indicate antagonism.
\end{itemize}
